Sub HelloWorld1()
    
    Dim strSql As String
    Dim tabName As String
    Dim Count As Long
    Dim i As Integer
    Dim j As Integer
    
    MsgBox "------------程序执行开始------------"
    DoCmd.SetWarnings False
    
    strSql = "select * into Sheet1 from Sheet"
    DoCmd.RunSQL strSql
    
    strSql = "select * into child10 from Sheet1 where 1=0"
    DoCmd.RunSQL strSql
    
    i = 0
    Count = CurrentDb.TableDefs("Sheet1").RecordCount
    j = Count / 9
     
    Do While (Count > 0)
        If (i < 10) Then
            i = i + 1
            Count = Count - j
            tabName = "child" & i
        End If
       
        strSql = "select top " & j & " * into " & tabName & " from Sheet1 order by ID"
        DoCmd.RunSQL strSql
        strSql = "delete from Sheet1 where ID in(select ID from " & tabName & ")"
        DoCmd.RunSQL strSql
        
    Loop
        
    If (Count <= 0) Then
        strSql = "drop table Sheet1"
        DoCmd.RunSQL strSql
    End If
    
    DoCmd.SetWarnings True
    MsgBox "--------------程序执行完毕，Sheet表数据已全部拆分！----------------"
End Sub
